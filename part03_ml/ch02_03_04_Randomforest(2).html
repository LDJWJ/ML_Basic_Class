def plot_feature_important_up(model, n_features, feature_names):
  imp = model.feature_importances_     # feature  
  plt.barh(range(n_features) , imp, align='center')  # (  )
  plt.yticks(np.arange(n_features), feature_names)  #y  
  plt.xlabel("feature importance")   # x ()
  plt.ylabel("feature")              # y 
  plt.ylim(-1, n_features)           # y
