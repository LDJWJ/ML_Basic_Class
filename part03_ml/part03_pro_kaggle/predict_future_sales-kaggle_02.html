def drop_duplicate(data, subset):
    print('()  :', data.shape)
    before = data.shape[0]
    data.drop_duplicates(subset,keep='first', inplace=True)
    data.reset_index(drop=True, inplace=True)
    print('()  :', data.shape)
    after = data.shape[0]
    print('Total Duplicate:', before-after)
